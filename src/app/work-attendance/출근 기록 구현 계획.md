// src/app/work-attendance/lib/api/real-attendance.api.ts
import axios, { AxiosResponse } from "axios";
import {
  AttendanceRes,
  ClockInReq,
  ClockOutReq,
  DailyAttendanceRes,
  ApiResponse,
} from "../types";

const BASE_URL =
  process.env.NEXT_PUBLIC_API_BASE_URL || "http://localhost:8080";

// Axios 인스턴스 생성
const apiClient = axios.create({
  baseURL: `${BASE_URL}/api/v1`,
  timeout: 10000,
  headers: {
    "Content-Type": "application/json",
  },
});

// 요청 인터셉터 (인증 토큰 자동 추가)
apiClient.interceptors.request.use(
  (config) => {
    const token = localStorage.getItem("accessToken");
    if (token) {
      config.headers.Authorization = `Bearer ${token}`;
    }
    return config;
  },
  (error) => Promise.reject(error)
);

// 응답 인터셉터 (에러 처리)
apiClient.interceptors.response.use(
  (response) => response,
  async (error) => {
    if (error.response?.status === 401) {
      // 토큰 만료 시 리프레시 시도
      await handleTokenRefresh();
      return apiClient.request(error.config);
    }
    return Promise.reject(error);
  }
);

/**
 * 실제 출근기록 API 서비스
 */
export class RealAttendanceApi {
  /**
   * 오늘의 출근 기록 조회
   */
  static async getTodayAttendance(): Promise<AttendanceRes> {
    try {
      const response: AxiosResponse<ApiResponse<AttendanceRes>> =
        await apiClient.get("/attendance/today");

      if (response.data.success) {
        return response.data.data;
      } else {
        throw new Error(response.data.message || "출근 기록 조회 실패");
      }
    } catch (error) {
      console.error("getTodayAttendance error:", error);
      throw this.handleApiError(error);
    }
  }

  /**
   * 출근 처리
   */
  static async clockIn(request: ClockInReq): Promise<AttendanceRes> {
    try {
      const response: AxiosResponse<ApiResponse<AttendanceRes>> =
        await apiClient.post("/attendance/clock-in", request);

      if (response.data.success) {
        return response.data.data;
      } else {
        throw new Error(response.data.message || "출근 처리 실패");
      }
    } catch (error) {
      console.error("clockIn error:", error);
      throw this.handleApiError(error);
    }
  }

  /**
   * 퇴근 처리
   */
  static async clockOut(request: ClockOutReq): Promise<AttendanceRes> {
    try {
      const response: AxiosResponse<ApiResponse<AttendanceRes>> =
        await apiClient.post("/attendance/clock-out", request);

      if (response.data.success) {
        return response.data.data;
      } else {
        throw new Error(response.data.message || "퇴근 처리 실패");
      }
    } catch (error) {
      console.error("clockOut error:", error);
      throw this.handleApiError(error);
    }
  }

  /**
   * API 에러 처리
   */
  private static handleApiError(error: any): Error {
    if (error.response) {
      // 서버 응답 에러
      const status = error.response.status;
      const message =
        error.response.data?.message || "서버 오류가 발생했습니다";

      switch (status) {
        case 400:
          return new Error(`잘못된 요청: ${message}`);
        case 401:
          return new Error("인증이 필요합니다");
        case 403:
          return new Error("권한이 없습니다");
        case 404:
          return new Error("요청한 리소스를 찾을 수 없습니다");
        case 500:
          return new Error("서버 내부 오류가 발생했습니다");
        default:
          return new Error(message);
      }
    } else if (error.request) {
      // 네트워크 에러
      return new Error("네트워크 연결을 확인해주세요");
    } else {
      // 기타 에러
      return new Error(error.message || "알 수 없는 오류가 발생했습니다");
    }
  }
}

### **1.2 React Query 훅 업데이트**

```typescript
// src/app/work-attendance/lib/hooks/use-real-attendance.ts
import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import { RealAttendanceApi } from "../api/real-attendance.api";
import { ClockInReq, ClockOutReq, AttendanceRes } from "../types";
import { useAppFeatures } from "./use-app-features";

/**
 * 실제 API를 사용하는 출근기록 훅
 */
export const useRealAttendance = () => {
  const queryClient = useQueryClient();
  const { errorFeedback, clockInFeedback, clockOutFeedback } = useAppFeatures();

  // 오늘의 출근 기록 조회
  const {
    data: attendance,
    isLoading,
    error,
    refetch,
  } = useQuery({
    queryKey: ["attendance", "today"],
    queryFn: RealAttendanceApi.getTodayAttendance,
    staleTime: 2 * 60 * 1000, // 2분
    cacheTime: 5 * 60 * 1000, // 5분
    retry: 3,
    retryDelay: (attemptIndex) => Math.min(1000 * 2 ** attemptIndex, 30000),
    onError: (error) => {
      console.error("출근 기록 조회 실패:", error);
      errorFeedback("데이터 로딩 실패", error.message);
    },
  });

  // 출근 처리 뮤테이션
  const clockInMutation = useMutation({
    mutationFn: RealAttendanceApi.clockIn,
    onSuccess: (data) => {
      // 캐시 업데이트
      queryClient.setQueryData(["attendance", "today"], data);
      clockInFeedback();
    },
    onError: (error: Error) => {
      console.error("출근 처리 실패:", error);
      errorFeedback("출근 처리 실패", error.message);
    },
  });

  // 퇴근 처리 뮤테이션
  const clockOutMutation = useMutation({
    mutationFn: RealAttendanceApi.clockOut,
    onSuccess: (data) => {
      // 캐시 업데이트
      queryClient.setQueryData(["attendance", "today"], data);
      clockOutFeedback();
    },
    onError: (error: Error) => {
      console.error("퇴근 처리 실패:", error);
      errorFeedback("퇴근 처리 실패", error.message);
    },
  });

  return {
    // 데이터
    attendance,

    // 상태
    isLoading:
      isLoading || clockInMutation.isLoading || clockOutMutation.isLoading,
    error,

    // 액션
    clockIn: (request: ClockInReq) => clockInMutation.mutateAsync(request),
    clockOut: (request: ClockOutReq) => clockOutMutation.mutateAsync(request),
    refreshData: refetch,

    // 개별 로딩 상태
    isClockingIn: clockInMutation.isLoading,
    isClockingOut: clockOutMutation.isLoading,
  };
};
```

// src/app/work-attendance/lib/realtime/attendance-sync.ts
export class AttendanceSync {
  private eventSource: EventSource | null = null;
  private queryClient: QueryClient;

  constructor(queryClient: QueryClient) {
    this.queryClient = queryClient;
  }

  /**
   * 실시간 동기화 시작
   */
  startSync() {
    const token = localStorage.getItem("accessToken");
    if (!token) return;

    this.eventSource = new EventSource(
      `${process.env.NEXT_PUBLIC_API_BASE_URL}/api/v1/attendance/stream?token=${token}`
    );

    this.eventSource.onmessage = (event) => {
      const data = JSON.parse(event.data);

      // 출근 기록 업데이트
      if (data.type === "ATTENDANCE_UPDATE") {
        this.queryClient.setQueryData(["attendance", "today"], data.attendance);
      }
    };

    this.eventSource.onerror = (error) => {
      console.error("SSE connection error:", error);
      // 재연결 로직
      setTimeout(() => this.startSync(), 5000);
    };
  }

  /**
   * 실시간 동기화 중지
   */
  stopSync() {
    if (this.eventSource) {
      this.eventSource.close();
      this.eventSource = null;
    }
  }
}

// src/app/work-attendance/lib/offline/attendance-queue.ts
interface QueuedAction {
  id: string;
  type: "CLOCK_IN" | "CLOCK_OUT";
  data: ClockInReq | ClockOutReq;
  timestamp: number;
}

export class AttendanceQueue {
  private queue: QueuedAction[] = [];
  private isProcessing = false;

  /**
   * 액션을 큐에 추가
   */
  enqueue(action: Omit<QueuedAction, "id" | "timestamp">) {
    const queuedAction: QueuedAction = {
      ...action,
      id: crypto.randomUUID(),
      timestamp: Date.now(),
    };

    this.queue.push(queuedAction);
    this.saveToStorage();

    // 온라인 상태면 즉시 처리
    if (navigator.onLine) {
      this.processQueue();
    }
  }

  /**
   * 큐 처리
   */
  async processQueue() {
    if (this.isProcessing || this.queue.length === 0) return;

    this.isProcessing = true;

    while (this.queue.length > 0) {
      const action = this.queue[0];

      try {
        if (action.type === "CLOCK_IN") {
          await RealAttendanceApi.clockIn(action.data as ClockInReq);
        } else {
          await RealAttendanceApi.clockOut(action.data as ClockOutReq);
        }

        // 성공 시 큐에서 제거
        this.queue.shift();
        this.saveToStorage();
      } catch (error) {
        console.error("Failed to process queued action:", error);
        break; // 실패 시 중단
      }
    }

    this.isProcessing = false;
  }

  private saveToStorage() {
    localStorage.setItem("attendanceQueue", JSON.stringify(this.queue));
  }

  loadFromStorage() {
    const stored = localStorage.getItem("attendanceQueue");
    if (stored) {
      this.queue = JSON.parse(stored);
    }
  }
}
