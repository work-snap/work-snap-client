# 🕐 Work Snap 아르바이트 출근 기록 시스템 기획서

## 📋 프로젝트 개요

### 1.1 프로젝트 소개

Work Snap은 아르바이트생의 출근 관리를 자동화하고 실시간으로 추적할 수 있는 **하이브리드 웹 애플리케이션**입니다. Flutter 기반 앱 패키징을 통해 네이티브 앱으로 배포 예정이며, 정확한 출근/퇴근 시간 기록과 스케줄 관리를 통해 근무자와 사업주 모두에게 투명하고 효율적인 근무 관리 경험을 제공합니다.

### 1.2 핵심 가치

- **정확성**: 실시간 출근/퇴근 시간 추적
- **투명성**: 모든 근무 기록의 명확한 표시
- **편의성**: 원터치 출근/퇴근 처리
- **유연성**: 다양한 근무 패턴 지원
- **크로스 플랫폼**: 웹과 모바일 앱 동시 지원

### 1.3 기술 스택

- **Backend**: Kotlin + Spring Boot 3.4.5
- **Frontend**: Next.js 15.3.3 + TypeScript + Tailwind CSS
- **Mobile**: Flutter (하이브리드 웹뷰)
- **Database**: MySQL 8.0
- **Authentication**: JWT + Kakao OAuth
- **HTTP Client**: Axios (토큰 자동 갱신 지원)

  attendance: AttendanceRes,
  currentTime: Date

  const now = currentTime.toTimeString().slice(0, 5); // HH:MM 형식
  const scheduledStart = attendance.scheduledStartTime;
  const scheduledEnd = attendance.scheduledEndTime;

  if (attendance.status === "SCHEDULED") {

        action: "CLOCK_OUT",

### 3.1 출근 관리 API (Axios 기반)

#### 3.1.1 기본 API 서비스

```typescript
// src/lib/attendance/attendance.ts
import api from "@/lib/api";

class AttendanceService {
  private baseUrl = "/api/attendance";

  // 출근 처리
  async clockIn(
    attendanceId: number,
    request: ClockInReq
  ): Promise<AttendanceRes> {
    const response = await api.post(
      `${this.baseUrl}/${attendanceId}/clock-in`,
      request
    );
    return response.data;
  }

  // 퇴근 처리
  async clockOut(
    attendanceId: number,
    request: ClockOutReq
  ): Promise<AttendanceRes> {
    const response = await api.post(
      `${this.baseUrl}/${attendanceId}/clock-out`,
      request
    );
    return response.data.data;
  }

  // 일별 출근 현황 조회
  async getDailyAttendance(date: string): Promise<DailyAttendanceRes> {
    const response = await api.get(`${this.baseUrl}/daily`, {
      params: { date },
    });
    return response.data.data;
  }

  // 진행 중인 출근 기록 조회
  async getActiveAttendance(): Promise<AttendanceRes[]> {
    const response = await api.get(`${this.baseUrl}/active`);
    return response.data.data;
  }
}

export const attendanceService = new AttendanceService();
```

#### 4.1.1 앱 브릿지 구현

```typescript
// src/lib/app-bridge/flutter-bridge.ts
class FlutterBridge {
  private isFlutterApp: boolean = false;

  constructor() {
    this.isFlutterApp = this.detectFlutterEnvironment();
  }

  private detectFlutterEnvironment(): boolean {
    return (
      !!(window as any).flutter_inappwebview || !!(window as any).flutterApp
    );
  }

  // 알림 표시 (alert 대체)
  async showNotification(
    title: string,
    message: string,
    type: "success" | "error" | "warning" | "info" = "info"
  ) {
    if (this.isFlutterApp) {
      return this.sendToFlutter({
        type: "SHOW_NOTIFICATION",
        data: { title, message, type },
      });
    } else {
      // 웹 환경에서는 토스트 표시
      this.showWebToast(title, message, type);
      return { success: true };
    }
  }

  // 진동 피드백
  async vibrate(pattern: number[] = [100]) {
    if (this.isFlutterApp) {
      return this.sendToFlutter({
        type: "VIBRATE",
        data: { pattern },
      });
    } else if (navigator.vibrate) {
      navigator.vibrate(pattern);
      return { success: true };
    }
    return { success: false, error: "Vibration not supported" };
  }

  // 위치 정보 요청
  async getCurrentLocation(): Promise<{
    latitude: number;
    longitude: number;
  } | null> {
    if (this.isFlutterApp) {
      const response = await this.sendToFlutter({
        type: "GET_LOCATION",
        data: {},
      });
      return response.success ? response.data : null;
    } else {
      return this.getWebLocation();
    }
  }

  private showWebToast(title: string, message: string, type: string) {
    const toast = document.createElement("div");
    toast.className = `fixed top-4 right-4 p-4 rounded-lg shadow-lg z-50 ${this.getToastColor(
      type
    )}`;
    toast.innerHTML = `
      <div class="font-semibold">${title}</div>
      <div class="text-sm">${message}</div>
    `;

    document.body.appendChild(toast);

    setTimeout(() => {
      toast.remove();
    }, 3000);
  }

  private getToastColor(type: string): string {
    switch (type) {
      case "success":
        return "bg-green-500 text-white";
      case "error":
        return "bg-red-500 text-white";
      case "warning":
        return "bg-yellow-500 text-black";
      default:
        return "bg-blue-500 text-white";
    }
  }

  get isApp(): boolean {
    return this.isFlutterApp;
  }
}

export const flutterBridge = new FlutterBridge();
```

// src/lib/app-bridge/use-app-features.ts
import { useEffect, useState } from "react";
import { flutterBridge } from "./flutter-bridge";

export const useAppFeatures = () => {
  const [isApp, setIsApp] = useState(false);
  const [appReady, setAppReady] = useState(false);

  useEffect(() => {
    setIsApp(flutterBridge.isApp);

    if (flutterBridge.isApp) {
      // 앱 준비 상태 확인
      flutterBridge
        .sendToFlutter({
          type: "APP_READY_CHECK",
          data: {},
        })
        .then((response) => {
          setAppReady(response.success);
        });
    } else {
      setAppReady(true);
    }
  }, []);

  // 출근 성공 피드백 (alert 대체)
  const clockInFeedback = async () => {
    await Promise.all([
      flutterBridge.showNotification(
        "출근 완료",
        "출근이 성공적으로 처리되었습니다",
        "success"
      ),
      flutterBridge.vibrate([100, 50, 100]),
    ]);
  };

  // 퇴근 성공 피드백 (alert 대체)
  const clockOutFeedback = async () => {
    await Promise.all([
      flutterBridge.showNotification(
        "퇴근 완료",
        "퇴근이 성공적으로 처리되었습니다",
        "success"
      ),
      flutterBridge.vibrate([100, 50, 100, 50, 100]),
    ]);
  };

  return {
    isApp,
    appReady,
    showNotification: flutterBridge.showNotification.bind(flutterBridge),
    vibrate: flutterBridge.vibrate.bind(flutterBridge),
    getCurrentLocation: flutterBridge.getCurrentLocation.bind(flutterBridge),
    clockInFeedback,
    clockOutFeedback,
  };
};

              퇴근 {attendance.actualEndTime}
            </div>
          )}
        </div>
      </div>

      {/* 출근/퇴근 타입 정보 */}
      {(attendance.clockInTypesKorean || attendance.clockOutTypesKorean) && (
        <div className="mb-4 p-3 bg-blue-50 rounded-lg text-center">
          <span className="text-blue-700 font-medium">
            {attendance.clockInTypesKorean || attendance.clockOutTypesKorean}
          </span>
        </div>
      )}

      {/* 버튼 미리보기 */}
      {isActionable && (
        <div className="mb-4 p-3 bg-gray-50 rounded-lg text-center">
          <div className="text-sm text-gray-600 mb-1">
            {isManualSelection ? "선택한 기능" : "AI 추천 기능"}
          </div>
          <div className="font-medium text-gray-800">
            <span className="mr-2">{currentButtonState.icon}</span>
            {currentButtonState.label}
          </div>
        </div>
      )}

      {/* 메인 액션 버튼 */}
      {attendance.status === "SCHEDULED" && (
        <button
          onClick={handleClockIn}
          disabled={isLoading || !appReady}
          className={`w-full py-4 px-4 rounded-lg font-medium text-white transition-all duration-200
            ${
              isLoading || !appReady
                ? "bg-gray-400 cursor-not-allowed"
                : `${currentButtonState.color} hover:opacity-90 active:scale-95`
            }`}
        >
          {isLoading ? (
            <div className="flex items-center justify-center">
              <div className="animate-spin rounded-full h-5 w-5 border-b-2 border-white mr-2"></div>
              처리 중...
            </div>
          ) : (
            <div className="flex items-center justify-center">
              <span className="mr-2">{currentButtonState.icon}</span>
              {currentButtonState.label}
            </div>
          )}
        </button>
      )}

      {attendance.status === "IN_PROGRESS" && (
        <button
          onClick={handleClockOut}
          disabled={isLoading || !appReady}
          className={`w-full py-4 px-4 rounded-lg font-medium text-white transition-all duration-200
            ${
              isLoading || !appReady
                ? "bg-gray-400 cursor-not-allowed"
                : `${currentButtonState.color} hover:opacity-90 active:scale-95`
            }`}
        >
          {isLoading ? (
            <div className="flex items-center justify-center">
              <div className="animate-spin rounded-full h-5 w-5 border-b-2 border-white mr-2"></div>
              처리 중...
            </div>
          ) : (
            <div className="flex items-center justify-center">
              <span className="mr-2">{currentButtonState.icon}</span>
              {currentButtonState.label}
            </div>
          )}
        </button>
      )}

      {attendance.status === "COMPLETED" && (
        <div className="text-center py-4">
          <div className="inline-flex items-center px-4 py-2 rounded-lg bg-green-100 text-green-800 font-medium">
            <span className="mr-2">✅</span>
            업무 완료
          </div>
        </div>
      )}

      {/* 앱 환경 표시 */}
      {isApp && (
        <div className="mt-4 text-center">
          <span className="inline-flex items-center px-2 py-1 rounded-full text-xs bg-purple-100 text-purple-700">
            <span className="mr-1">📱</span>
            모바일 앱
          </span>
        </div>
      )}
    </div>
  );

  function getStatusColor(status: string): string {
    switch (status) {
      case "SCHEDULED":
        return "bg-gray-100 text-gray-800";
      case "IN_PROGRESS":
        return "bg-blue-100 text-blue-800";
      case "COMPLETED":
        return "bg-green-100 text-green-800";
      case "ABSENT":
        return "bg-red-100 text-red-800";
      default:
        return "bg-gray-100 text-gray-800";
    }
  }
}

// src/lib/hooks/use-smart-button.ts
import { useState, useEffect, useMemo, useCallback } from "react";
import {
  AttendanceRes,
  ClockInType,
  ClockOutType,
} from "@/lib/attendance/types";

interface ButtonOption {
  type: ClockInType | ClockOutType;
  label: string;
  icon: string;
  color: string;
}

interface SmartButtonState extends ButtonOption {
  action: "CLOCK_IN" | "CLOCK_OUT";
  variant: "default" | "early" | "late" | "overtime";
  isRecommended: boolean;
}

const BUTTON_OPTIONS = {
  CLOCK_IN: [
    {
      type: "NORMAL" as ClockInType,
      label: "출근하기",
      icon: "🏢",
      color: "bg-main",
    },
    {
      type: "EARLY_ARRIVAL" as ClockInType,
      label: "조기출근하기",
      icon: "⏰",
      color: "bg-blue-500",
    },
  ],
  CLOCK_OUT: [
    {
      type: "NORMAL" as ClockOutType,
      label: "퇴근하기",
      icon: "🏠",
      color: "bg-green-500",
    },
    {
      type: "EARLY_DEPARTURE" as ClockOutType,
      label: "조퇴하기",
      icon: "🚪",
      color: "bg-orange-500",
    },
    {
      type: "LATE_DEPARTURE" as ClockOutType,
      label: "연장근무 종료",
      icon: "⏰",
      color: "bg-red-500",
    },
  ],
};

export const useSmartButton = (attendance: AttendanceRes) => {
  const [currentTime, setCurrentTime] = useState(new Date());
  const [selectedIndex, setSelectedIndex] = useState(0);
  const [isManualSelection, setIsManualSelection] = useState(false);
  const [autoResetTimer, setAutoResetTimer] = useState<NodeJS.Timeout | null>(
    null
  );

  // 1분마다 현재 시간 업데이트
  useEffect(() => {
    const interval = setInterval(() => {
      setCurrentTime(new Date());
    }, 60000);

    return () => clearInterval(interval);
  }, []);

  // 시간 비교 유틸리티
  const timeToMinutes = (timeStr: string): number => {
    const [hours, minutes] = timeStr.split(":").map(Number);
    return hours * 60 + minutes;
  };

  const getCurrentMinutes = (): number => {
    const now = new Date();
    return now.getHours() * 60 + now.getMinutes();
  };

  // AI 추천 버튼 상태 계산
  const recommendedState = useMemo((): SmartButtonState => {
    const currentMinutes = getCurrentMinutes();
    const scheduledStartMinutes = timeToMinutes(attendance.scheduledStartTime);
    const scheduledEndMinutes = timeToMinutes(attendance.scheduledEndTime);

    if (attendance.status === "SCHEDULED") {
      const earlyThreshold = 15; // 15분 전까지 조기출근

      if (currentMinutes < scheduledStartMinutes - earlyThreshold) {
        return {
          action: "CLOCK_IN",
          type: "EARLY_ARRIVAL",
          label: "조기출근하기",
          icon: "⏰",
          color: "bg-blue-500",
          variant: "early",
          isRecommended: true,
        };
      } else {
        return {
          action: "CLOCK_IN",
          type: "NORMAL",
          label: "출근하기",
          icon: "🏢",
          color: "bg-main",
          variant: "default",
          isRecommended: true,
        };
      }
    } else if (attendance.status === "IN_PROGRESS") {
      const earlyLeaveThreshold = 30; // 30분 전까지 조퇴
      const overtimeThreshold = 10; // 10분 후부터 연장근무

      if (currentMinutes < scheduledEndMinutes - earlyLeaveThreshold) {
        return {
          action: "CLOCK_OUT",
          type: "EARLY_DEPARTURE",
          label: "조퇴하기",
          icon: "🚪",
          color: "bg-orange-500",
          variant: "early",
          isRecommended: true,
        };
      } else if (currentMinutes > scheduledEndMinutes + overtimeThreshold) {
        return {
          action: "CLOCK_OUT",
          type: "LATE_DEPARTURE",
          label: "연장근무 종료",
          icon: "⏰",
          color: "bg-red-500",
          variant: "overtime",
          isRecommended: true,
        };
      } else {
        return {
          action: "CLOCK_OUT",
          type: "NORMAL",
          label: "퇴근하기",
          icon: "🏠",
          color: "bg-green-500",
          variant: "default",
          isRecommended: true,
        };
      }
    }

    // 기본값
    return {
      action: "CLOCK_IN",
      type: "NORMAL",
      label: "출근하기",
      icon: "🏢",
      color: "bg-main",
      variant: "default",
      isRecommended: false,
    };
  }, [attendance, currentTime]);

  // 현재 선택된 버튼 상태
  const currentButtonState = useMemo((): SmartButtonState => {
    if (isManualSelection) {
      const action =
        attendance.status === "SCHEDULED" ? "CLOCK_IN" : "CLOCK_OUT";
      const options = BUTTON_OPTIONS[action];
      const selectedOption = options[selectedIndex];

      return {
        ...selectedOption,
        action,
        variant: "default",
        isRecommended: false,
      };
    }
    return recommendedState;
  }, [isManualSelection, selectedIndex, recommendedState, attendance.status]);

  // 버튼 옵션 순환
  const cycleButtonOption = useCallback(() => {
    const action = attendance.status === "SCHEDULED" ? "CLOCK_IN" : "CLOCK_OUT";
    const options = BUTTON_OPTIONS[action];
    const nextIndex = (selectedIndex + 1) % options.length;

    setSelectedIndex(nextIndex);
    setIsManualSelection(true);

    // 기존 타이머 해제
    if (autoResetTimer) {
      clearTimeout(autoResetTimer);
    }

    // 5분 후 자동으로 AI 추천 모드로 돌아가기
    const timer = setTimeout(() => {
      setIsManualSelection(false);
      setSelectedIndex(0);
    }, 5 * 60 * 1000);

    setAutoResetTimer(timer);
  }, [selectedIndex, attendance.status, autoResetTimer]);

  // 컴포넌트 언마운트 시 타이머 정리
  useEffect(() => {
    return () => {
      if (autoResetTimer) {
        clearTimeout(autoResetTimer);
      }
    };
  }, [autoResetTimer]);

  return {
    currentButtonState,
    cycleButtonOption,
    isManualSelection,
    recommendedState,
    currentTime,
    timeUntilWorkStart: Math.max(
      0,
      timeToMinutes(attendance.scheduledStartTime) - getCurrentMinutes()
    ),
    timeUntilWorkEnd: Math.max(
      0,
      timeToMinutes(attendance.scheduledEndTime) - getCurrentMinutes()
    ),
  };
};
